Installation

Step 1: Link required files

<!-- vendor js -->
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.11.0/jquery.min.js"></script>

<!-- chaffle js -->
<script src="./dist/js/jquery.chaffle.min.js"></script>

Step 2: HTML markup

<a href="#" class="chaffle" data-lang="en">Engligh</a>
<a href="#" class="chaffle" data-lang="ja">日本語（漢字）</a>
<a href="#" class="chaffle" data-lang="ja-hiragana">ひらがな</a>
<a href="#" class="chaffle" data-lang="ja-katakana">カタカナ</a>

Step 3: Call the chaffle

$(document).ready(function() {
  $('.chaffle').chaffle({
    speed: 20,
    time: 140
  });
});
