Setup

<!-- Vendor js -->
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.11.0/jquery.min.js"></script>

<!-- chaffle js -->
<script src="./js/chaffle.min.js"></script>
<script>
  
  $(document).ready(function() {
    $('.chaffle').chaffle();
  }); 
  
</script>
